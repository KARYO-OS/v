#!/bin/bash
# ============================================================
# KARYO OS Phase 2 Database Sync - Local Machine Only
# ============================================================
# 
# This script runs on your LOCAL machine (not in dev container)
# to sync Phase 2 migrations to Supabase production.
#
# Prerequisites:
# - PostgreSQL client (psql) installed
# - Network access to Supabase
# - .env.local with credentials
#
# Usage: bash sync-db-local.sh
# ============================================================

set -e

# Extract credentials from .env.local
PROJECT_REF="upvrcaivdzuxozdwzvdq"
DB_PASSWORD="Kedinasan2020"
DB_URL="postgresql://postgres:${DB_PASSWORD}@db.${PROJECT_REF}.supabase.co:5432/postgres"

echo "╔════════════════════════════════════════════════════════╗"
echo "║  KARYO OS Phase 2 Database Sync                        ║"
echo "║  (Run from LOCAL machine, not in container)            ║"
echo "╚════════════════════════════════════════════════════════╝"
echo ""

# Check psql
if ! command -v psql &> /dev/null; then
  echo "❌ psql not found"
  echo "Install PostgreSQL client:"
  echo "  macOS: brew install postgresql"
  echo "  Ubuntu: sudo apt install postgresql-client"
  exit 1
fi

# Check migration files
MIGRATIONS=(
  "supabase/migrations/20260420152000_phase2_multisatuan_foundation.sql"
  "supabase/migrations/20260420154000_phase2_multisatuan_rls.sql"
  "supabase/migrations/20260420160000_phase2_multisatuan_rpc_dual_support.sql"
)

echo "📋 Checking migrations..."
for migration in "${MIGRATIONS[@]}"; do
  if [ ! -f "$migration" ]; then
    echo "❌ File not found: $migration"
    echo "Make sure you're in the project root directory"
    exit 1
  fi
done
echo "✅ All migration files found"
echo ""

# Test connection
echo "🔗 Testing database connection..."
if ! psql "$DB_URL" -c "SELECT 1" > /dev/null 2>&1; then
  echo "❌ Cannot connect to database"
  echo "Check credentials and network connectivity"
  exit 1
fi
echo "✅ Database connected"
echo ""

# Execute migrations
echo "⏳ Executing Phase 2 migrations..."
echo ""

FAILED=0

for i in "${!MIGRATIONS[@]}"; do
  migration="${MIGRATIONS[$i]}"
  stage=$((i + 1))
  
  stage_name="Foundation"
  [ $stage -eq 2 ] && stage_name="RLS Hardening"
  [ $stage -eq 3 ] && stage_name="RPC Dual-Support"
  
  echo "📍 Stage $stage: $stage_name"
  
  if psql "$DB_URL" -f "$migration" > /dev/null 2>&1; then
    echo "   ✅ Executed successfully"
  else
    echo "   ❌ FAILED"
    FAILED=1
    break
  fi
  echo ""
done

if [ $FAILED -eq 0 ]; then
  echo "╔════════════════════════════════════════════════════════╗"
  echo "║          ✅ DATABASE SYNC SUCCESSFUL ✅                ║"
  echo "╚════════════════════════════════════════════════════════╝"
  echo ""
  echo "Next steps:"
  echo "  1. Wait ~2 minutes for GitHub Actions to deploy frontend"
  echo "  2. Login to app with admin account"
  echo "  3. Navigate to: Admin → Satuan"
  echo "  4. Verify multi-unit management works"
  echo ""
else
  echo "╔════════════════════════════════════════════════════════╗"
  echo "║             ❌ SYNC FAILED ❌                          ║"
  echo "╚════════════════════════════════════════════════════════╝"
  exit 1
fi
